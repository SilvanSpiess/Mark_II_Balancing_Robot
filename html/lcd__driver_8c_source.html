101static void displayReset(void)
102{
103 chipSelect(INSTRUCTION_REGISTER);
104 HAL_GPIO_WritePin(LCD_RESET_GPIO_Port, LCD_RESET_Pin, GPIO_PIN_RESET);
105 HAL_Delay(1);
106 HAL_GPIO_WritePin(LCD_RESET_GPIO_Port, LCD_RESET_Pin, GPIO_PIN_SET);
107 chipSelect(DATA_REGISTER);

139void lcd_init(void)
140{
141 lcd_clear();
142 displayReset();
143
144 sendInstruction(ADC_NORMAL);
145 sendInstruction(DISPLAY_OFF);
146 sendInstruction(COMMON_OUTPUT_MODE_REVERSE);
147 sendInstruction(BIAS_ONE_NINTH);

154void lcd_setPixel(uint8_t xPosition, uint8_t yPosition, bool pixelIsSet)
155{
156 if ((xPosition < NUMBER_OF_ROWS) && (yPosition < NUMBER_OF_COLUMNS)) {
157 if (pixelIsSet)
158 displayBuffer[xPosition].rawData |= (1 << yPosition);
159 else
160 displayBuffer[xPosition].rawData &= ~(1 << yPosition);
161 }
162}

172void lcd_setChar(uint8_t xPosition, uint8_t yPosition, unsigned char charToSet, lcd_fontSize size, bool contrastIsInverted)
173{
174#define ASCII_TABLE_OFFSET ((uint8_t) 32)
175
176 charToSet -= ASCII_TABLE_OFFSET;
177
178 for (uint8_t columnCounter = 0; columnCounter < fontSizes[size].height; columnCounter++)
179 for (uint8_t rowCounter = 0; rowCounter < fontSizes[size].width; rowCounter++){
180 if (lcd_font6x8[charToSet * fontSizes[size].height + columnCounter] & (1 << rowCounter))
181 lcd_setPixel(xPosition + rowCounter, yPosition + columnCounter, !contrastIsInverted);
182 else
183 lcd_setPixel(xPosition + rowCounter, yPosition + columnCounter, contrastIsInverted);
184 }
185}

198void lcd_setLine(int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint8_t state)
199{
200 int16_t i, x[2], y[2];
201 float xF, yF;
202
203 xF = x1, x[1] = 0;
204 yF = y1, y[1] = 0;
205
206 x[0] = abs(x2-x1);
207 y[0] = abs(y2-y1);
208
209 if(x[0]) x[1] = (x2-x1)/x[0];
210 if(y[0]) y[1] = (y2-y1)/y[0];
211
212 if(x[0] >= y[0]) {
213 for(i = 0; i<x[0]; i++) {
214 if( xF >= 0 && yF >= 0) lcd_setPixel(xF, yF, state);
215 xF += x[1];
216 yF += ((float)y[0]/x[0])*y[1];
217 }
218 }
219 else {
220 for(i = 0; i<y[0]; i++) {
221 if( xF >= 0 && yF >= 0) lcd_setPixel(xF, yF, state);
222 xF += ((float)x[0]/y[0])*x[1];
223 yF += y[1];
224 }
225 }
226}

229void lcd_setRect(uint16_t x, uint16_t y, uint16_t w, uint16_t h, uint8_t state)

237 // bottom horizontal line
238 for( j=0; j<w; j++ ) {

241 // left vertical line

243 lcd_setPixel((x), y+j, state);
244 }
245 // right vertical line
246 for( j=0; j<h; j++ ) {
247 lcd_setPixel((x)+w, y+j, state);
248 }

255void lcd_show(void){
256 for (pages pageIndex = PAGE_0; pageIndex < NUMBER_OF_PAGES; pageIndex++){
257 //$dnd added (instruction)
258 sendInstruction((instruction) (PAGE_ADDRESS_SET_0 + pageIndex));
259 sendInstruction(COLUMN_ADDRESS_SET_BIT);
260 sendInstruction(BOOSTER_RATIO_SET);
261 for (uint8_t rowIndex = 0; rowIndex < NUMBER_OF_ROWS; rowIndex++)
262 sendData(displayBuffer[rowIndex].pageSet[pageIndex]);
263 }
264}
